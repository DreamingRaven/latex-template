  Equations:

  \begin{equation}
    \label{sigmoid}
    \sigma(x) = \frac{1}{1+e^{-x}}
  \end{equation}

  \begin{equation}
    \label{sigmoid_approx}
    \sigma(x) \approx 0.5 + 0.197x + -0.004x^3
  \end{equation}

  \begin{equation}
    \label{cnn_activation}
    a^{<t>}=\sigma(w_{i}^{<t>}x^{<t>}+b_i^{<t>})
  \end{equation}

  \begin{equation}
    \label{gradient}
    \frac{df}{d\sigma} = (1-\sigma{x}) * \sigma{x}
  \end{equation}

  \begin{equation}
    \label{weight_update}
    % learning rate
    w_i^{j+1<t>} = w_i^{j<t>} - (l * \frac{df}{dw_i^{j<t>}})
  \end{equation}
